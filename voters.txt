           FROM public.wbw_voters
          GROUP BY wbw_voters.street
        ), running AS (
         SELECT row_number() OVER (ORDER BY roads_count.total_voters DESC, roads_count.street) AS "row",
            roads_count.street,
            roads_count.total_voters AS num_of_voters,
            sum(roads_count.total_voters) OVER (ORDER BY roads_count.total_voters DESC, roads_count.street) AS running_count,
            round((((roads_count.total_voters)::numeric / sum(roads_count.total_voters) OVER ()) * (100)::numeric), 2) AS pct_voters
           FROM roads_count
        )
 SELECT running."row",
    running.street,
    running.num_of_voters,
    running.running_count,
    running.pct_voters,
    sum(running.pct_voters) OVER (ORDER BY running.pct_voters DESC, running.street) AS running_pct
   FROM running
  WITH NO DATA;
